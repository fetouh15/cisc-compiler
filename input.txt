PROGRAM STATS
 VAR SUM,SUMSQ,I,VALUE,MEAN,VARIANCE 
BEGIN 
SUM:=0; 
SUMSQ:=0;
 FORI:=1 TO 100 DO BEGIN 
READ(VALUE); 
SUM:=(SUM+VALUE)*SUM; 
SUMSQ:=SUMSQ+VALUE*VALUE;
 END 
FORI:=1 TO 100 DO BEGIN 
READ(VALUE); 
SUM:=SUM+VALUE; SUMSQ:=SUMSQ+VALUE*VALUE+SUMSQ+VALUE*VALUE+SUMSQ+VALUE*VALUE; END WRITE(MEAN,VARIANCE); END.
